

## 知识点

-   多维数组

    -   添加元素`res = append(res, []int{n1, n2, n3})`

-   排序`sort.Ints(nums)`



#### [15. 三数之和](https://leetcode.cn/problems/3sum/)

方法一：排序 + 双指针

```go
func threeSum(nums []int) [][]int {
    sort.Ints(nums)
    res := [][]int{}

    for i := 0; i < len(nums) - 2; i++ {
        n1 := nums[i]

        // 排序后第一位为正数，即所有数都为正数
        if n1 > 0 {
            break
        }

        // 三元组首位去重
        if i > 0 && n1 == nums[i - 1] {
            continue
        }

        l, r := i + 1, len(nums) - 1
        for l < r {
            n2, n3 := nums[l], nums[r]
            if n1 + n2 + n3 == 0 {
                res = append(res, []int{n1, n2, n3})
                for l < r && n2 == nums[l]{
                    l++
                }
                for l < r && n3 == nums[r] {
                    r--
                }
            } else if n1 + n2 + n3 < 0 {
                l++
            } else {
                r--
            }
        }
    }
    return res
}
```
