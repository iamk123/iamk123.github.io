### [5. 最长回文子串](https://leetcode.cn/problems/longest-palindromic-substring/)

#### 方法一：中心扩散法

```java
class Solution {
    public String longestPalindrome(String s) {
        String res = "";
        for(int i = 0; i < s.length(); i++) {
            String s1 = extend(s, i, i);
            res = s1.length() > res.length() ? s1 : res; // 以i为中心，奇数情况		aba

            String s2 = extend(s, i, i + 1);
            res = s2.length() > res.length() ? s2 : res;	// 以i和i+1为中心，偶数情况 	abba
        }
        
        return res;
    }

    public String extend(String s, int i, int j) {
        String tmp = "";
        while(i >= 0 && j < s.length() && s.charAt(i) == s.charAt(j)) {
            if(j - i + 1 > tmp.length()) {
                tmp = s.substring(i, j + 1);
            }
            i--;
            j++;
        }

        return tmp;
    }
}
```



#### 方法二：dp

```java
class Solution {
    public String longestPalindrome(String s) {
        int len = s.length();
        boolean[][] dp = new boolean[len][len];

        int left = 0;
        int right = 0;
        int maxLen = 0;
        for(int i = len - 1; i >= 0; i--) {
            for(int j = i; j < len; j++) {
                if(s.charAt(i) == s.charAt(j)) {
                    if(j - i <= 1) {    // a  aa 
                        dp[i][j] = true;
                    } else {    // aba  abca
                        dp[i][j] = dp[i + 1][j - 1];
                    }
                }

                if(dp[i][j] && j - i + 1 > maxLen) {
                    left = i;
                    right = j;
                    maxLen = j - i + 1;
                }
            }
        }

        // for(boolean[] d : dp) {
        //     System.out.println(Arrays.toString(d));
        // }

        return s.substring(left, right + 1);
    }
}
```

```java
class Solution {
    public String longestPalindrome(String s) {
        int n = s.length();
        if(n <= 1) return s;
        String res = String.valueOf(s.charAt(0));	// a ac
        int[][] dp = new int[n][n];

        for(int i = n - 1; i >= 0; i--) {
            dp[i][i] = 1;
            for(int j = i + 1; j < n; j++) {
                if(s.charAt(i) == s.charAt(j)) {
                    if(j - i <= 1) {
                        dp[i][j] = 1;
                    } else {
                        dp[i][j] = dp[i + 1][j - 1];
                    }
                } 

                if(dp[i][j] == 1 && j - i + 1 > res.length()) {
                    res = s.substring(i, j + 1);

                }
            }
        }

        }

        return res;
    }
}
```
